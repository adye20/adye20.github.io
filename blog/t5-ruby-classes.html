      map {|object| block} => array
    

      >> teams = ['patriots', 'colts' 'seahawks', 'packers']

      >> teams.map {|team| teams.capitalize}
      => ["Patriots", "Colts", "Seahawks", "Packers"]
      # returns an array with capitalized strings

      >> team.each {|team| puts + ' are going to the conference championships'}
      => Patriots are going to the conference championships
      => Colts are going to the conference championships
      => Seahawks are going to the conference championships
      => Packers are going to the conference championships
      # simply outputs each element in the array with added string
    

      >> array = [6,7,8]
      >> array.map {|x| x*x}
      => [36, 49, 64]

      >> cities = ['chicago', 'miami', 'honolulu']
      >> cities.map(&:capitalize.sort)
      => ['Chicago', 'Miami', 'Honolulu']
